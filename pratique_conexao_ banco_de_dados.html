<?php
$servidor = "seu_servidor";
$usuario = "seu_usuario";
$senha = "sua_senha";
$banco = "seu_banco_de_dados";

// Estabelecer a conexão
$conexao = mysqli_connect($servidor, $usuario, $senha, $banco);
// Verificar a conexão
if (!$conexao) {
  die("Conexão falhou: " . mysqli_connect_error());
}
echo "Conexão bem-sucedida";
// Fechar a conexão (não é estritamente necessário, pois o PHP fecha automaticamente a conexão ao final do script)
mysqli_close($conexao);
?>
