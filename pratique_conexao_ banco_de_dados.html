Código PHP de Exemplo:

<?php
// Definição das informações de conexão
$servidor = "localhost"; // Endereço do servidor do banco de dados (normalmente é "localhost")
$usuario = "root"; // Nome de usuário do banco de dados
$senha = ""; // Senha do banco de dados
$dbname = "db_trailerv1.0"; // Nome do banco de dados que você deseja se conectar

// Conexão com o banco de dados
$conexao = mysqli_connect($servidor, $usuario, $senha, $dbname);

// Verificação da conexão
if (!$conexao) {
    die("Conexão falhou: " . mysqli_connect_error());
}
?>
